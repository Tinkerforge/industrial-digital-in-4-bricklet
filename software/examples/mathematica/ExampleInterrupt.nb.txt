Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="cno"(* Change to your UID *)

(* Create IPConnection and device object *)
ipcon=NETNew["Tinkerforge.IPConnection"]
idi4=NETNew["Tinkerforge.BrickletIndustrialDigitalIn4",uid,ipcon]
ipcon@Connect[host,port]

(* Enable interrupt on pin 0 *)
idi4@SetInterrupt[1]

(* Callback function for interrupts *)
InterruptCB[sender_,interruptMask_,valueMask_]:=
 Module[{},
  Print["Interrupt by: "<>StringJoin[Map[ToString,IntegerDigits[interruptMask,2,4]]]];
  Print["Value: "<>StringJoin[Map[ToString,IntegerDigits[valueMask,2,4]]]]
 ]

AddEventHandler[idi4@Interrupt,InterruptCB]

Input["Click OK to exit"]

(* Clean up *)
ipcon@Disconnect[]
ReleaseNETObject[idi4]
ReleaseNETObject[ipcon]
